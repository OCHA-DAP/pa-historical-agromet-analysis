## Admin 0

```python
# Due to the difference in CRS, will have to change Niger shapefiles to raster CRS
input_raster = rasterio.open(file_path)
array = input_raster.read(1)
print(ner_shp_adm0.crs == input_raster.crs)
ner_shp_adm0_crs = ner_shp_adm0.to_crs(input_raster.crs.to_dict())
ner_shp_adm0_clipped_crs = ner_shp_adm0_clipped.to_crs(input_raster.crs.to_dict())
```

```python
ner_shp_adm0.drop(
    ["OBJECTID", "Shape_Leng", "Shape_Area", "ISO2"], axis=1, inplace=True
)
for rfe in rfe_sources:
    if rfe == "noaa_cpc":
        file_dir = noaa_wrsi_dir
    else:
        file_dir = chirps_wrsi_dir
    adm0_wrsi_df = pd.DataFrame()
    for area in areas:
        file_list = [
            file for file in os.listdir(file_dir / area) if file.endswith("do.tif")
        ]
        for file in file_list:
            file_path = Path(file_dir / area / file)
            input_raster = rasterio.open(file_path)
            array = input_raster.read(1)
            array = np.where(array > 200, 254, array)
            summary_stats = zonal_stats(
                ner_shp_adm0_crs,
                array,
                stats=["mean", "median"],
                nodata=254,
                all_touched=True,
                affine=input_raster.transform,
            )
            adm0_stats = pd.DataFrame(summary_stats)
            file_df = pd.merge(
                ner_shp_adm0_crs[["ISO3", "NOMPAY"]],
                adm0_stats,
                left_index=True,
                right_index=True,
            )
            time_period = file[(file.find("w") + 1) : file.find("do.tif")]
            if rfe == "noaa_cpc":
                date_obj = (
                    "20"
                    + time_period[0:2]
                    + "-"
                    + f"{math.ceil(int(time_period[2:4]) / 3):02}"
                )
                dekad_obj = time_period[2:4]
            else:
                date_obj = (
                    time_period[0:4]
                    + "-"
                    + f"{math.ceil(int(time_period[4:6]) / 3):02}"
                )
                dekad_obj = time_period[4:6]
            file_df["month"] = date_obj
            file_df["dekad"] = dekad_obj
            file_df["land_use"] = area
            adm0_wrsi_df = pd.concat([adm0_wrsi_df, file_df], axis=0)
    adm0_wrsi_df.to_csv(wrsi_dir / str("ner_adm0_" + rfe + "_wrsi.csv"), index=False)
```

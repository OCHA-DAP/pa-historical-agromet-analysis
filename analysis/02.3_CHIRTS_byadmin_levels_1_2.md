```python
data_dir = Path(os.getenv("HAM_DIR")) / "temperature/monthly_chirts/"
output_dir = Path(os.getenv("HAM_DIR")) / "temperature/"
shp_dir = Path(os.getenv("AA_DATA_DIR")) / "public/raw/ner/cod_ab/ner_cod_ab/"
```

```python
ner_shp_adm0 = gpd.read_file(shp_dir / "adm0.shp.zip")

```

```python
# clipped adm1 and adm2
ner_shp_adm0_clipped = gpd.read_file(shp_dir_clp / "ner_codab_adm0_clipped_17N.gpkg")
ner_shp_adm1_clipped = gpd.read_file(shp_dir_clp / "ner_codab_adm1_clipped_17N.gpkg")
```

```python
ner_shp_adm0.total_bounds
```

## Admin 0

```python
ner_shp_adm0.drop(
    ["OBJECTID", "Shape_Leng", "Shape_Area", "ISO2"], axis=1, inplace=True
)
adm0_chirts_df = pd.DataFrame()
for file in file_list:
    file_path = Path(data_dir / file)
    input_raster = rasterio.open(file_path)
    array = input_raster.read(1)
    summary_stats = zonal_stats(
        ner_shp_adm0,
        array,
        stats=["mean", "median"],
        nodata=0,
        all_touched=True,
        affine=input_raster.transform,
    )
    adm0_stats = pd.DataFrame(summary_stats)
    file_df = pd.merge(
        ner_shp_adm0[["ISO3", "NOMPAY"]],
        adm0_stats,
        left_index=True,
        right_index=True,
    )
    file_df["date"] = (
        file[(file.find("CHIRTSmax.") + 10) : file.find(".tif")]
    ).replace(".", "-")
    adm0_chirts_df = pd.concat([adm0_chirts_df, file_df], axis=0)
adm0_chirts_df.to_csv(output_dir / "ner_adm0_chirts.csv", index=False)
```

```python
    input_raster = rasterio.open(file_path)
    array = input_raster.read(1)
    summary_stats = zonal_stats(

        array,

        nodata=0,
        all_touched=True,
        affine=input_raster.transform,
```

```python
        ner_adm2_clipped_geom[["rowcacode2", "NOMDEP"]],

    file_df2 = pd.merge(
        ner_shp_adm2[["rowcacode2", "NOMDEP"]],
        file_df,
        how="left",

    file_df2["date"] = (
```

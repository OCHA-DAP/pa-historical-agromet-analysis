```python
def aggregating_admin(file_list, geodf, admID1, admID2):
    adm_chirts_df = pd.DataFrame()
    for file in file_list:
        file_path = Path(data_dir / file)
        input_raster = rasterio.open(file_path)
        array = input_raster.read(1)
        summary_stats = zonal_stats(
            geodf,
            array,
            stats=["mean", "median"],
            nodata=0,
            all_touched=True,
            affine=input_raster.transform,
        )
        adm_stats = pd.DataFrame(summary_stats)
        file_df = pd.merge(
            geodf[[admID1, admID2]],
            adm_stats,
            left_index=True,
            right_index=True,
        )
        file_df["date"] = (
            file[(file.find("CHIRTSmax.") + 10) : file.find(".tif")]
        ).replace(".", "-")
        adm_chirts_df = pd.concat([adm_chirts_df, file_df], axis=0)
    return adm_chirts_df
```

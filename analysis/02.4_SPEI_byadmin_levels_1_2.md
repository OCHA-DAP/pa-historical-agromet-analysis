## Admin 0

```python
ner_shp_adm0.drop(
    ["OBJECTID", "Shape_Leng", "Shape_Area", "ISO2"], axis=1, inplace=True
)
adm0_spei_df = pd.DataFrame()
for mon in months:
    spei_arr = spei_data.sel(time=mon).values
    summary_stats = zonal_stats(
        ner_shp_adm0,
        spei_arr,
        stats=["mean", "median"],
        nodata=-9999,
        all_touched=True,
        affine=spei_affine,
    )
    adm0_stats = pd.DataFrame(summary_stats)
    file_df = pd.merge(
        ner_shp_adm0[["ISO3", "NOMPAY"]],
        adm0_stats,
        left_index=True,
        right_index=True,
    )
    file_df["date"] = f"{mon.year:04}" + "-" + f"{mon.month:02}"
    adm0_spei_df = pd.concat([adm0_spei_df, file_df], axis=0)
adm0_spei_df.to_csv(data_dir / "ner_adm0_spei.csv", index=False)
```

adm2_spei_df = pd.DataFrame()
